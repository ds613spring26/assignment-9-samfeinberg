  1. Sentiment Analysis

    • Filter joy/positive words from each of the three sentiment lexicons:

      • nrc (joy)
      • afinn (positive scores ≥ 1)
      • bing (positive)
nrc_joy <- get_sentiments("nrc") |>
  filter(sentiment == "joy")

affin_positive <- get_sentiments("afinn") |>
  filter(value > 0)

positive_words <- get_sentiments("bing") |>
  filter(sentiment == "positive") |>
  left_join(nrc_joy, by = "word") |>
  left_join(affin_positive, by = "word") |>
  rename(
    pos_or_neg = sentiment.x,
    sentiment = sentiment.y
  )

    • Join each with Emma’s text and:

positive_words_emma <- austen_tokenized |>
  filter(book == "Emma") |>
  inner_join(positive_words, by = "word", relationship = "many-to-many") |>
  count(word, sort = TRUE) |>
  filter(n > 50)
